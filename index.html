Usage (LESS)

#mdgrid > .create(desktop, 12, 60px, 20px);


// large desktops
@media (min-width: 1221px) {

	// 12x80 + 12x20 = 1200
	#mdgrid > .create(desktop, 12, 80px, 20px);

}

// tablet & small desktops
@media (min-width: 768px) and (max-width: 979px) {


	// 12x44 + 12x20 = 768
	#mdgrid > .create(desktop, 12, 44px, 20px);

	// 6x108 + 6x20 = 768px
	#mdgrid > .create(tablet, 6, 108px, 20px);


}

// phones
@media screen and (max-width: 767px) {


	// 12x30 + 12x10 = 480px
	#mdgrid > .create(desktop, 12, 20px, 20px);

	// 2x220 + 2x20 = 480px
	#mdgrid > .create(mobile, 2, 220px, 20px);


}

// phones old
@media screen and (max-width: 500px) {

	// 12x20 + 12x10 = 360
	#mdgrid > .create(desktop, 12, 10px, 20px);

	// 2x160 + 2x20 = 360px
	#mdgrid > .create(mobile, 2, 160px, 20px);

}
