Basic grid setup for a 960 grid (less)

				
@prefix: column;
@column-count: 12;
@column-width: 60px;
@gutter-width: 20px;

.mdgrid(@prefix, @column-count, @column-width, @gutter-width);

LESS

				
// default: 12x60 + 12x20 = 960
.mdgrid(desktop, 12, 60px, 20px);

// large desktops
@media (min-width: 1221px) {
	// 12x80 + 12x20 = 1200
	.mdgrid(desktop, 12, 80px, 20px);
}

// tablet & small desktops
@media (min-width: 768px) and (max-width: 979px) {

	// 6x108 + 6x20 = 768px
	.mdgrid(tablet, 6, 108px, 20px);
}

// phones
@media screen and (max-width: 767px) {

	// 2x220 + 2x20 = 480px
	.mdgrid(mobile, 2, 220px, 20px);
}

// phones old
@media screen and (max-width: 500px) {

	// 2x160 + 2x20 = 360px
	.mdgrid(mobile, 2, 160px, 20px);

}
